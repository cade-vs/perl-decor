Описание на клавишите за навигация

клавиш " { " прелистване до първа страница

клавиш " } " прелистване до последна страница

клавиш " [ " прелистване на предишна страница

клавиш " ] " прелистване на следваща страница

клавиш " + " показване на повече редове на една страница

клавиш " - " показване на по-малко редове на една страница

клавиш " = " показване на всички редове в една страница (ограничено)

клавиш " * " връщане на стандартния бров редове на страница
